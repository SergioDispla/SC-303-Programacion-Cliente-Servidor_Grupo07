# Crear tabla registro de ventas de productos
CREATE TABLE IF NOT EXISTS ventaproductos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cedula VARCHAR(20),
    totalpagado FLOAT,
    tipopago VARCHAR(10)
);

# Crear tabla registro de ventas de servicios
CREATE TABLE IF NOT EXISTS ventaservicios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cedula VARCHAR(20),
    totalpagado FLOAT,
    tipopago VARCHAR(10)
);


#Crear tabla productos
CREATE TABLE IF NOT EXISTS productos (
    codigo INT PRIMARY KEY,
    nombre VARCHAR(20),
    descripcion VARCHAR(255),
    precio float,
    cant_stock INT
);

#Crear tabla registro mantenimientos
CREATE TABLE IF NOT EXISTS registromantenimientos (
    id INT AUTO_INCREMENT PRIMARY KEY,
	cedula VARCHAR(20),
    placa VARCHAR(10),
	servicio VARCHAR(255),
	id_operario VARCHAR(10)
	
	
);

#Crear tabla clientes
CREATE TABLE clientes (
    cedula VARCHAR(20),
    nombre VARCHAR(100),
    telefono VARCHAR(20),
    direccion VARCHAR(100),
    correoElectronico VARCHAR(100),
    fechaRegistro DATE
);

#Crear tabla vehiculos
CREATE TABLE vehiculos (
    placa VARCHAR(10) PRIMARY KEY,
    marca VARCHAR(100),
    modelo VARCHAR(100),
    año INT,
    kilometraje INT,
	cedula VARCHAR(20)
);

#Crear tabla operarios
CREATE TABLE operarios (
	id_operario VARCHAR(10) PRIMARY KEY,
    cedula VARCHAR(20), 
    nombre VARCHAR(100),
    telefono VARCHAR(20),
    direccion VARCHAR(100),
    correoElectronico VARCHAR(100),
	salario float,
    fechaContratacion DATE
);
